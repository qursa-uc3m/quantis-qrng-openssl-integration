cmake_minimum_required(VERSION 3.16)

project(custom_rng_provider VERSION 1.1.0 LANGUAGES C)

include(CTest)

set(CMAKE_C_STANDARD 99)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS ON)
set(CMAKE_C_VISIBILITY_PRESET hidden)
set(CMAKE_POSITION_INDEPENDENT_CODE ON)

option(DEBUG "Build with debug information" OFF)
option(XOR_RANDOM "XOR Quantis output with the operating-system RNG" ON)
option(MEASURE_RNG "Record the number of generated bytes in shared memory" OFF)
option(ALLOW_OS_FALLBACK "Use the operating-system RNG if Quantis is unavailable" ON)

set(DEVICE_TYPE "USB" CACHE STRING "Quantis device type (USB or PCIE)")
set_property(CACHE DEVICE_TYPE PROPERTY STRINGS USB PCIE)
set(QUANTIS_LIB "NO" CACHE STRING "Use the Quantis library (YES or NO)")
set_property(CACHE QUANTIS_LIB PROPERTY STRINGS YES NO)
set(DEVICE_NUMBER "0" CACHE STRING "Quantis device number")
set(QUANTIS_DEVICE_PATTERN "/dev/qrandom%d" CACHE STRING
    "printf-style device path used when QUANTIS_LIB=NO")
set(QUANTIS_LIB_DIR "/opt/quantis/Libs-Apps" CACHE PATH
    "Path to the Quantis library installation")

if(NOT DEVICE_TYPE STREQUAL "USB" AND NOT DEVICE_TYPE STREQUAL "PCIE")
    message(FATAL_ERROR "DEVICE_TYPE must be USB or PCIE")
endif()
if(NOT QUANTIS_LIB STREQUAL "YES" AND NOT QUANTIS_LIB STREQUAL "NO")
    message(FATAL_ERROR "QUANTIS_LIB must be YES or NO")
endif()
if(NOT DEVICE_NUMBER MATCHES "^[0-9]+$")
    message(FATAL_ERROR "DEVICE_NUMBER must be a non-negative integer")
endif()
if(NOT QUANTIS_DEVICE_PATTERN MATCHES "%d")
    message(FATAL_ERROR "QUANTIS_DEVICE_PATTERN must contain %d")
endif()

find_package(OpenSSL 3.0 REQUIRED COMPONENTS Crypto)

add_library(custom_qrng_provider SHARED
    quantis_qrng_provider.c
    quantis_qrng_provider_rand.c
)

target_compile_definitions(custom_qrng_provider PRIVATE
    QUANTIS_PROV_PKG_VERSION="${PROJECT_VERSION}"
    DEVICE_NUMBER=${DEVICE_NUMBER}
    QUANTIS_DEVICE_PATTERN="${QUANTIS_DEVICE_PATTERN}"
)
target_compile_options(custom_qrng_provider PRIVATE
    $<$<C_COMPILER_ID:GNU,Clang>:-Wall;-Wextra;-Wpedantic;-Wconversion>
)
target_link_libraries(custom_qrng_provider PRIVATE OpenSSL::Crypto)

if(DEBUG)
    target_compile_definitions(custom_qrng_provider PRIVATE DEBUG)
    target_compile_options(custom_qrng_provider PRIVATE
        $<$<C_COMPILER_ID:GNU,Clang>:-g>
    )
endif()
if(XOR_RANDOM)
    target_compile_definitions(custom_qrng_provider PRIVATE XOR_RANDOM)
endif()
if(ALLOW_OS_FALLBACK)
    target_compile_definitions(custom_qrng_provider PRIVATE ALLOW_OS_FALLBACK)
endif()
if(MEASURE_RNG)
    target_compile_definitions(custom_qrng_provider PRIVATE MEASURE_RNG)
    if(CMAKE_SYSTEM_NAME STREQUAL "Linux")
        target_link_libraries(custom_qrng_provider PRIVATE rt)
    endif()
endif()

if(QUANTIS_LIB STREQUAL "YES")
    if(DEVICE_TYPE STREQUAL "USB")
        target_compile_definitions(custom_qrng_provider PRIVATE DEVICE_USB)
    else()
        target_compile_definitions(custom_qrng_provider PRIVATE DEVICE_PCIE)
    endif()

    find_path(QUANTIS_INCLUDE_DIR Quantis.h
        HINTS "${QUANTIS_LIB_DIR}/Quantis")
    find_library(QUANTIS_LIBRARY NAMES Quantis
        HINTS "${QUANTIS_LIB_DIR}/build/Quantis")
    if(NOT QUANTIS_INCLUDE_DIR OR NOT QUANTIS_LIBRARY)
        message(FATAL_ERROR
            "Quantis headers or library not found below ${QUANTIS_LIB_DIR}")
    endif()
    target_compile_definitions(custom_qrng_provider PRIVATE USE_QUANTIS_READ)
    target_include_directories(custom_qrng_provider PRIVATE "${QUANTIS_INCLUDE_DIR}")
    target_link_libraries(custom_qrng_provider PRIVATE "${QUANTIS_LIBRARY}")
endif()

set_target_properties(custom_qrng_provider PROPERTIES
    OUTPUT_NAME custom_qrng_provider
)

if(BUILD_TESTING)
    add_executable(provider_integration_test tests/provider_integration_test.c)
    target_link_libraries(provider_integration_test PRIVATE OpenSSL::Crypto)
    if(ALLOW_OS_FALLBACK)
        target_compile_definitions(provider_integration_test PRIVATE
            EXPECT_GENERATE_SUCCESS)
    endif()
    target_compile_options(provider_integration_test PRIVATE
        $<$<C_COMPILER_ID:GNU,Clang>:-Wall;-Wextra;-Wpedantic;-Wconversion>
    )
    add_test(NAME provider.integration COMMAND provider_integration_test)
    set_tests_properties(provider.integration PROPERTIES
        ENVIRONMENT "OPENSSL_MODULES=$<TARGET_FILE_DIR:custom_qrng_provider>"
    )
endif()

install(TARGETS custom_qrng_provider LIBRARY DESTINATION lib)

set(CPACK_PACKAGE_VENDOR
    "QURSA project, Telematics Engineering Department, GAST Group, UC3M")
set(CPACK_PACKAGE_VERSION "${PROJECT_VERSION}")
set(CPACK_DEBIAN_PACKAGE_MAINTAINER
    "QURSA project, Telematics Engineering Department, GAST Group, UC3M")
include(CPack)
